        private void timer1_Tick(object sender, EventArgs e)
        {
            if (int.Parse(count.Text) == 0)
            {
                if (Red.Checked)
                {
                    Green.Checked = true;

                    return;
                }
                if (Green.Checked)
                {
                  Yellow.Checked = true;
                    return;
                }
                if (Yellow.Checked)
                {
                    Red.Checked = true;
                    return;
                }
            }
            else
            {
                count.Text = (int.Parse(count.Text) - 1).ToString();
            }
            if (int.Parse(count.Text) == 5 && Red.Checked == true) KY.Checked = true;
        }
